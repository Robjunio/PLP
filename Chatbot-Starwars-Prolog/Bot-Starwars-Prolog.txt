:- initialization(bot_jedi).

talk :-
    write('O que deseja saber? '), nl,
    read(Topic),
    interpret(Topic).
    
% Main.
bot_jedi :-
    write('Started'), nl,
    talk.

interpret(fim) :-
    write('end').
    
interpret(planeta) :-
    write('Qual planeta deseja conhecer? '),
    read(Planet),
    planet(Planet, Answer), !,
    write(Answer),
    talk.

interpret(nave) :-
    write('Qual nave deseja conhecer? '),
    read(Ship),
    ship(Ship, Answer), !,
    write(Answer), nl,
    talk.

interpret(mandaloriano) :-
    write('insira o nome de um personagem: '),
    read(Mandalorian),
    mandalorian(Mandalorian, Answer), !,
    write(Answer), nl,
    talk.

interpret(forca) :-
    write('insira o nome de um personagem: '),
    read(Force),
    force(Force, Answer), !,
    write(Answer), nl,
    talk.

interpret(citacao) :-
    write('De qual personagem deseja conhecer uma citacao? '),
    read(Quote),
    quote(Quote, Answer), !,
    write(Answer), nl,
    talk.

interpret(pais) :-
    write('insira o nome do pai/mae para descobrir o filho: '),
    read(Parent),
    parent(Parent, Answer), !,
    write(Answer), nl,
    talk.

interpret(irmao) :-
    write('insira o nome do personagem para descobrir o seu irmao: '),
    read(Siblin),
    siblin(Siblin, Answer), !,
    write(Answer), nl,
    talk.

interpret(mestre) :-
    write('insira o nome do personagem para descobrir o seu aluno: '),
    read(Master),
    master(Master, Answer), !,
    write(Answer), nl,
    talk.

interpret(droid) :-
    write('Qual droid deseja conhecer? '),
    read(Droid),
    droid(Droid, Answer), !,
    write(Answer), nl,
    talk.

interpret(_) :-
    write('Não conheco esse topico, mas posso te ajudar com alguma outra coisa?'), nl,
    talk.
